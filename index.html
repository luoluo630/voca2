        // --- TTS Helper Functions (Required for PCM conversion) ---

 

        /** Converts a Base64 string to an ArrayBuffer. */

        const base64ToArrayBuffer = (base64) => {

            const binaryString = atob(base64);

            const len = binaryString.length;

            const bytes = new Uint8Array(len);

            for (let i = 0; i < len; i++) {

                bytes[i] = binaryString.charCodeAt(i);

            }

            return bytes.buffer;

        };

 

        /** Writes a string to a DataView at a given offset. */

        const writeString = (view, offset, string) => {

            for (let i = 0; i < string.length; i++) {

                view.setUint8(offset + i, string.charCodeAt(i));

            }

        };

 

        /** Converts PCM (Pulse Code Modulation) audio data to a playable WAV Blob. */

        const pcmToWav = (pcm16, sampleRate = 24000) => {

            const numChannels = 1;

            const bytesPerSample = 2; // 16-bit PCM

 

            // Create WAV header buffer (44 bytes)

            const buffer = new ArrayBuffer(44 + pcm16.byteLength);

            const view = new DataView(buffer);

            let offset = 0;

 

            // RIFF chunk descriptor

            writeString(view, offset, 'RIFF'); offset += 4;

            view.setUint32(offset, 36 + pcm16.byteLength, true); offset += 4;

            writeString(view, offset, 'WAVE'); offset += 4;

 

            // fmt sub-chunk

            writeString(view, offset, 'fmt '); offset += 4;

            view.setUint32(offset, 16, true); offset += 4; // Sub-chunk size

            view.setUint16(offset, 1, true); offset += 2; // Audio format (1 = PCM)

            view.setUint16(offset, numChannels, true); offset += 2;

            view.setUint32(offset, sampleRate, true); offset += 4;

            view.setUint32(offset, sampleRate * numChannels * bytesPerSample, true); offset += 4; // Byte rate

            view.setUint16(offset, numChannels * bytesPerSample, true); offset += 2; // Block align

            view.setUint16(offset, 16, true); offset += 2; // Bits per sample (16)

 

            // data sub-chunk

            writeString(view, offset, 'data'); offset += 4;

            view.setUint32(offset, pcm16.byteLength, true); offset += 4;

 

            // Write PCM data

            const pcmBytes = new Uint8Array(buffer, offset);

            pcmBytes.set(new Uint8Array(pcm16.buffer));

 

            // Convert ArrayBuffer to Blob

            return new Blob([buffer], { type: 'audio/wav' });

        };

        // --- TTS API Call ---

 

        /** Generates and plays audio for a given text using the Gemini TTS model. */

        const speak = async (text, element) => {

            element.classList.add('speaking-animation');

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key=${apiKey}`;

            const voiceName = "Puck"; // Upbeat, friendly voice

 

            const payload = {

                contents: [{

                    parts: [{ text: text }]

                }],

                generationConfig: {

                    responseModalities: ["AUDIO"],

                    speechConfig: {

                        voiceConfig: {

                            prebuiltVoiceConfig: { voiceName: voiceName }

                        }

                    }

                },

                model: "gemini-2.5-flash-preview-tts"

            };

 

            const maxRetries = 3;

            let delay = 1000;

 

            for (let i = 0; i < maxRetries; i++) {

                try {

                    showMessage("Generating audio...", true);

                    

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

 

                    if (!response.ok) {

                        throw new Error(`HTTP error! status: ${response.status}`);

                    }

 

                    const result = await response.json();

                    const part = result?.candidates?.[0]?.content?.parts?.[0];

                    const audioData = part?.inlineData?.data;

                    const mimeType = part?.inlineData?.mimeType;

 

                    if (audioData && mimeType && mimeType.startsWith("audio/L16")) {

                        // The API returns raw signed PCM 16 bit audio data.

                        const pcmData = base64ToArrayBuffer(audioData);

                        const pcm16 = new Int16Array(pcmData);

                        

                        // The default sample rate for L16 is 24000 Hz for this model

                        const sampleRate = 24000;

                        

                        const wavBlob = pcmToWav(pcm16, sampleRate);

                        const audioUrl = URL.createObjectURL(wavBlob);

 

                        const audio = new Audio(audioUrl);

                        audio.onended = () => {

                            element.classList.remove('speaking-animation');

                            hideMessage();

                            URL.revokeObjectURL(audioUrl); // Clean up

                        };

                        audio.onerror = () => {

                            element.classList.remove('speaking-animation');

                            showMessage("Error playing audio.", false);

                            setTimeout(hideMessage, 2000);

                        };

                        await audio.play();

                        return; // Success

                    } else {

                        throw new Error("Invalid or missing audio data in response.");

                    }

 

                } catch (error) {

                    console.error(`TTS API attempt ${i + 1} failed:`, error);

                    if (i < maxRetries - 1) {

                        await new Promise(resolve => setTimeout(resolve, delay));

                        delay *= 2; // Exponential backoff

                    } else {

                        showMessage("Failed to generate audio after multiple retries.", false);

                        element.classList.remove('speaking-animation');

                        setTimeout(hideMessage, 3000);

                    }

                }

            }
